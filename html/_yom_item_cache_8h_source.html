62 void GetOrLoad(const InKeyType& pKey, const FYomCacheLoadDelegate& pLoadCallback, const FYomCacheReturnDelegate& pReturnCallback)
63 {
64 FScopeLock lock(&mLock);

67#if YOM_CACHE_DIAGNOSTICS

119 void CheckCacheSize()
120 {
121 if (mCachedItems.Num() < InCacheSize) return;
122 FScopeLock lock(&mLock);
123 InKeyType lowestKey = {};
124 float lowestTiming = 0;
125 for (auto item: mCachedItems)
126 {
127 if (item.Value.Value >= lowestTiming) continue;
128 lowestTiming = item.Value.Value;
129 lowestKey = item.Key;
130 }
131 mCachedItems.Remove(lowestKey);
